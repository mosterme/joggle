21   * @author  $Author$

24  public class SearchServlet extends HttpServlet {
25  
26  	private static final long serialVersionUID = 1146058744384688900L;
27  	private static final Logger log = LoggerFactory.getLogger(SearchServlet.class);
28  	private static final Manager manager = Manager.getInstance();
29  
30  	@SuppressWarnings("rawtypes")
31  	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
32  		long t0 = System.currentTimeMillis();
33  		String callback = request.getParameter("$callback");
34  		String url = Serializer.decode(request.getRequestURI());
35  		List list = null;
36  		if (url.contains("/type/")) {
37  			String type = FilenameUtils.getName(url);
38  			list = manager.byType(type);
39  		}
40  		else if (url.contains("/artist/")) {
41  			String artist = FilenameUtils.getName(url);
42  			if ("".equals(artist)) list = manager.artists();
43  			else list = manager.byArtist(artist);
44  		}
45  		else if (url.contains("/album/")) {
46  			String album = FilenameUtils.getName(url);
47  			if ("".equals(album)) list = manager.albums();
48  			else list = manager.byAlbum(album);
49  		}
50  		else if (url.contains("/song/")) {
51  			String song = FilenameUtils.getName(url);
52  			if ("".equals(song)) list = manager.songs();
53  			else list = Collections.singletonList(manager.find(song));
54  		}
55  		else {
56  			String keywords = url.split("/")[3];
57  			list = manager.search(keywords);
58  		}
59  		String result = "{\"d\":" + Serializer.toJson(list) + "}";
60  		if (callback != null) result = callback + "(" + result + ")";
61  		response.setCharacterEncoding("utf-8");
62  		Writer out = response.getWriter(); out.write(result);
63  		response.setStatus(HttpServletResponse.SC_OK);
64  		long t1 = System.currentTimeMillis();
65  		if (log.isInfoEnabled()) {
66  			String string = "search: " + format(url, 40) + " duration: " + (t1 - t0) + "ms" + "\tresults: " + list.size();
67  			if (log.isDebugEnabled()) string += "\tcallback: " + callback;
68  			log.info(string);
69  		}
70  	}
71  
72  	private static String format(String s, int m) {
73  		if (log.isDebugEnabled()) return s;
74  		String r = s.replaceFirst("/joggle/search/([^/]+)/(.+)$", "$1..$2")
75  		.replaceFirst("^album", "album.")
76  		.replaceFirst("^type",  "type..")
77  		+ "............................";
78  		return r.substring(0, m - 1) + ".";
79  	}
80  }
