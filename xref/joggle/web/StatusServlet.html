25  public class StatusServlet extends HttpServlet {
26  
27  	private static final long serialVersionUID = 3640618359057072513L;
28  	private static final SimpleDateFormat df = new SimpleDateFormat("d MMM yyyy HH:mm:ss");
29  
30  	@Override
31  	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
32  		Writer out = response.getWriter();
33  		out.write(Serializer.toJson(status()));
34  	}
35  
36  	@SuppressWarnings({ "rawtypes", "unchecked" })
37  	private static Map status() {
38  		Manager manager = Manager.getInstance();
39  		int i = manager.albums().size(), j = manager.artists().size(), k = manager.songs().size();
40  		Map result = new TreeMap();
41  		result.put("application.revision", manager.getProperty("joggle.application.revision"));
42  		result.put("application.timestamp", manager.getProperty("joggle.application.timestamp"));
43  		result.put("application.version", manager.getProperty("joggle.application.version"));
44  		result.put("manager.albums", i);
45  		result.put("manager.artists", j);
46  		result.put("manager.songs", k);
47  		Runtime rt = Runtime.getRuntime();
48  		result.put("memory.free", FileUtils.byteCountToDisplaySize(rt.freeMemory()));
49  		result.put("memory.max", FileUtils.byteCountToDisplaySize(rt.maxMemory()));
50  		result.put("memory.total", FileUtils.byteCountToDisplaySize(rt.totalMemory()));
51  		RuntimeMXBean rb = ManagementFactory.getRuntimeMXBean();
52  		result.put("vm.started", df.format(rb.getStartTime()));
53  		result.put("vm.uptime", rb.getUptime()); // TODO: format this
54  		// OperatingSystemMXBean os = ManagementFactory.getOperatingSystemMXBean();
55  		// result.put("system.load", os.getSystemLoadAverage());
56  		return result;
57  	}
58  }
