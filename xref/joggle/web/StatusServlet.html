23   * @author  $Author$

26  public class StatusServlet extends HttpServlet {
27  
28  	private static final long serialVersionUID = 3640618359057072513L;
29  
30  	@Override
31  	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
32  		Writer out = response.getWriter();
33  		out.write(Serializer.toJson(status()));
34  	}
35  
36  	@SuppressWarnings({ "rawtypes", "unchecked" })
37  	private static Map status() {
38  		Manager manager = Manager.getInstance();
39  		int i = manager.albums().size(), j = manager.artists().size(), k = manager.songs().size();
40  		Map result = new TreeMap();
41          SimpleDateFormat df = new SimpleDateFormat("d MMM yyyy HH:mm:ss");
42  		result.put("application.revision", manager.getProperty("joggle.application.revision"));
43  		result.put("application.timestamp", manager.getProperty("joggle.application.timestamp"));
44  		result.put("application.version", manager.getProperty("joggle.application.version"));
45  		result.put("manager.albums", i);
46  		result.put("manager.artists", j);
47  		result.put("manager.songs", k);
48  		Runtime rt = Runtime.getRuntime();
49  		result.put("memory.free", FileUtils.byteCountToDisplaySize(rt.freeMemory()));
50  		result.put("memory.max", FileUtils.byteCountToDisplaySize(rt.maxMemory()));
51  		result.put("memory.total", FileUtils.byteCountToDisplaySize(rt.totalMemory()));
52  		RuntimeMXBean rb = ManagementFactory.getRuntimeMXBean();
53  		result.put("vm.started", df.format(rb.getStartTime()));
54  		result.put("vm.uptime", rb.getUptime()); // TODO: format this
55  		OperatingSystemMXBean os = ManagementFactory.getOperatingSystemMXBean();
56  		result.put("system.load", os.getSystemLoadAverage());
57  		return result;
58  	}
59  }
