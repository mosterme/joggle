20  public class Scanner {

22  	private static final Logger log = LoggerFactory.getLogger(Scanner.class);
23  	private final FileFilter filter = new AudioFileFilter(true);
24  	private long bytes, directories, files;
25  
26  	public void scan(String s) {
27  		log.info("Scanning " + s);
28  		long t0 = System.currentTimeMillis();
29  		scan(new File(s));
30  		long t1 = System.currentTimeMillis();
31  		log.info("> Scan report for " + s);
32  		log.info(">   audio files   " + files);
33  		log.info(">   direcories    " + directories);
34  		log.info(">   scanned bytes " + FileUtils.byteCountToDisplaySize(bytes));
35  		log.info(">   time elapsed  " + (t1 - t0) + " millis");
36  	}
37  
38  	public void scan(File d) {
39  		if (d.isDirectory()) {
40  			if (log.isDebugEnabled()) log.debug("scanning " + d);
41  			File[] fa = d.listFiles(filter);
42  			for (File f : fa) scan(f);
43  			directories++;
44  		}
45  		else if (d.isFile()) {
46  			handle(d);
47  			files++; bytes += d.length();
48  		}
49  	}
50  
51  	private void handle(File f) {
52  		try {
53  			AudioFile af = AudioFileIO.read(f); Tag tag = af.getTag();
54  			String type = af.getAudioHeader().getEncodingType().substring(0, 3).toLowerCase();
55  			if (tag != null) {
56  				Manager manager = Manager.getInstance();
57  				String artist = tag.getFirst(FieldKey.ARTIST);
58  				String album = tag.getFirst(FieldKey.ALBUM);
59  				String title = tag.getFirst(FieldKey.TITLE);
60  				String file = f.getAbsolutePath();
61  				String id = Serializer.hash(file);
62  				String t = tag.getFirst(FieldKey.TRACK);
63  				Integer track = t.matches("[0-9]+") ? Integer.parseInt(t) : null;
64  				String genre = tag.getFirst(FieldKey.GENRE);
65  				if (genre.matches("\\(?[0-9]{1,3}\\)?")) genre = manager.getGenre(genre.replace("(", "").replace(")", ""));
66  				String albumArtist = tag.getFirst(FieldKey.ALBUM_ARTIST);
67  				if (brokenEncoding(artist, f)) artist = fixEncoding(artist);
68  				if (brokenEncoding(albumArtist, f)) albumArtist = fixEncoding(albumArtist);
69  				if (brokenEncoding(album, f)) album = fixEncoding(album);
70  				if (brokenEncoding(title, f)) title = fixEncoding(title);
71  				Boolean artwork = tag.getFirstArtwork() != null && tag.getFirstArtwork().getBinaryData() != null; // has artwork but no data?
72  				if (artwork && log.isDebugEnabled()) log.debug("Found artwork in " + f);
73  				Song s = new Song(id, type, artist, album, track, title, genre, artwork, file, System.currentTimeMillis(), albumArtist);
74  				manager.merge(s);
75  			}
76  			else {
77  				log.warn("No meta tag found in " + f);
78  			}
79  		}
80  		catch (Throwable t) {
81  			log.warn(t.getMessage());
82  			log.warn(f.getAbsolutePath());
83  		}
84  	}
85

87  	private boolean brokenEncoding(String s, File f) {
88  		boolean b = s.contains("Ã");
89  		if (b) log.info("Non utf-8 header in " + f.getAbsolutePath());
90  		return b;
91  	}
92  
93  	private String fixEncoding(String s) throws UnsupportedEncodingException {
94  		String r = new String(s.getBytes("iso-8859-1"), "utf-8");
95  		if (log.isDebugEnabled()) log.debug("fixEncoding: " + s + " -> " + r);
96  		return r;
97  	}
